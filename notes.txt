// Fast hex to RGB
function h2r(hex) {
  const h = hex.replace('#', '');
  return [
    parseInt(h.slice(0, 2), 16),
    parseInt(h.slice(2, 4), 16),
    parseInt(h.slice(4, 6), 16)
  ];
}

// Convert RGB to LAB color space for perceptual distance
function rgb2lab(r, g, b) {
  // Normalize RGB
  r /= 255; g /= 255; b /= 255;
  
  // Apply gamma correction
  r = r > 0.04045 ? Math.pow((r + 0.055) / 1.055, 2.4) : r / 12.92;
  g = g > 0.04045 ? Math.pow((g + 0.055) / 1.055, 2.4) : g / 12.92;
  b = b > 0.04045 ? Math.pow((b + 0.055) / 1.055, 2.4) : b / 12.92;
  
  // Convert to XYZ
  let x = (r * 0.4124 + g * 0.3576 + b * 0.1805) * 100;
  let y = (r * 0.2126 + g * 0.7152 + b * 0.0722) * 100;
  let z = (r * 0.0193 + g * 0.1192 + b * 0.9505) * 100;
  
  // Normalize for D65 illuminant
  x /= 95.047; y /= 100.000; z /= 108.883;
  
  // Convert to LAB
  x = x > 0.008856 ? Math.pow(x, 1/3) : (7.787 * x) + 16/116;
  y = y > 0.008856 ? Math.pow(y, 1/3) : (7.787 * y) + 16/116;
  z = z > 0.008856 ? Math.pow(z, 1/3) : (7.787 * z) + 16/116;
  
  return [
    (116 * y) - 16,  // L
    500 * (x - y),   // a
    200 * (y - z)    // b
  ];
}

// Perceptual color distance using CIEDE2000-like calculation
function dist(hex1, hex2) {
  const [r1, g1, b1] = h2r(hex1);
  const [r2, g2, b2] = h2r(hex2);
  
  const [L1, a1, b1Lab] = rgb2lab(r1, g1, b1);
  const [L2, a2, b2Lab] = rgb2lab(r2, g2, b2);
  
  // Simplified CIEDE2000 (more weight on lightness and hue)
  const dL = L1 - L2;
  const da = a1 - a2;
  const db = b1Lab - b2Lab;
  
  return Math.sqrt(dL*dL*2 + da*da + db*db);
}

// Find closest Tailwind color
function findClosest(hex) {
  let best = null;
  let bestDist = Infinity;
  
  for (const [family, shades] of Object.entries(TW)) {
    for (const [shade, twHex] of Object.entries(shades)) {
      const d = dist(hex, twHex);
      if (d < bestDist) {
        bestDist = d;
        best = { family, shade, hex: twHex, dist: d };
      }
    }
  }
  
  return best;
}

// Main popup logic
document.addEventListener('DOMContentLoaded', async () => {
  const status = document.getElementById('status');
  const list = document.getElementById('list');
  
  try {
    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
    
    if (!tab.url.includes('coolors.co')) {
      status.textContent = 'Please open a Coolors.co page';
      list.innerHTML = '<div class="empty"><div class="empty-icon">üé®</div><div class="empty-text">Navigate to coolors.co to use this extension</div><button id="openCoolors" class="action-btn" style="margin-top: 16px;">Coolors.co Trending Palettes</button></div>';
      
      // Add click handler for the button
      document.getElementById('openCoolors').addEventListener('click', () => {
        chrome.tabs.create({ url: 'https://coolors.co/palettes/trending', active: true });
      });
      
      return;
    }
    
    // Function to extract palettes from page
    async function extractPalettes() {
      const results = await chrome.scripting.executeScript({
        target: { tabId: tab.id },
        func: () => {
          const palettes = [];
          
          // Check if we're on trending/explore page with multiple palettes
          const paletteCards = document.querySelectorAll('.palette-card');
          
          if (paletteCards.length > 0) {
            // Multiple palettes (trending page)
            paletteCards.forEach(card => {
              const title = card.querySelector('.palette-card_name')?.textContent.trim() || 'Untitled Palette';
              const colorDivs = card.querySelectorAll('.palette-card_colors > div span');
              const colors = [];
              
              colorDivs.forEach(span => {
                const hex = span.textContent.trim();
                if (/^[0-9A-Fa-f]{6}$/.test(hex)) {
                  colors.push('#' + hex.toLowerCase());
                }
              });
              
              if (colors.length > 0) {
                palettes.push({ title, colors });
              }
            });
          } else {
            // Single palette page
            const hexes = new Set();
            
            // Method 1: Get from generator_color_hex divs
            document.querySelectorAll('.generator_color_hex').forEach(el => {
              const hex = el.textContent.trim();
              if (/^[0-9A-Fa-f]{6}$/.test(hex)) {
                hexes.add('#' + hex.toLowerCase());
              }
            });
            
            // Method 2: Search text content for hex patterns
            const text = document.body.innerText || '';
            const matches = text.match(/\b[0-9A-Fa-f]{6}\b/g);
            if (matches) {
              matches.forEach(m => {
                if (!/[G-Zg-z]/.test(m)) {
                  hexes.add('#' + m.toLowerCase());
                }
              });
            }
            
            // Method 3: Check URL for palette hexes
            const urlMatch = location.pathname.match(/\/([0-9a-fA-F-]+)/);
            if (urlMatch && urlMatch[1]) {
              urlMatch[1].split('-').forEach(part => {
                if (/^[0-9A-Fa-f]{6}$/.test(part)) {
                  hexes.add('#' + part.toLowerCase());
                }
              });
            }
            
            if (hexes.size > 0) {
              palettes.push({ title: 'Current Palette', colors: Array.from(hexes) });
            }
          }
          
          return palettes;
        }
      });
      
      return results[0].result;
    }
    
    // Function to render palettes
    function renderPalettes(palettes) {
      if (!palettes || palettes.length === 0) {
        status.textContent = 'No colors found';
        list.innerHTML = '<div class="empty"><div class="empty-icon">üîç</div><div class="empty-text">No hex colors detected on this page. Scroll to load more palettes!</div></div>';
        return;
      }
      
      const totalColors = palettes.reduce((sum, p) => sum + p.colors.length, 0);
      status.innerHTML = `<span style="color: #00c48c;">‚óè</span> Found ${palettes.length} palette${palettes.length === 1 ? '' : 's'} (${totalColors} colors) - Live updating`;
      
      // Store current scroll position
      const scrollPos = list.scrollTop;
      
      list.innerHTML = palettes.map((palette, idx) => {
        const colorRows = palette.colors.map(hex => {
          const match = findClosest(hex);
          const className = `${match.family}-${match.shade}`;
          
          return `
            <div class="color-row">
              <div class="swatch" style="background: ${hex}"></div>
              <div class="info">
                <div class="tw-name">${className}</div>
                <div class="details">${hex} ‚Üí ${match.hex}</div>
              </div>
              <button class="copy-btn" data-copy="${className}">Copy</button>
            </div>
          `;
        }).join('');
        
        return `
          <div class="palette-group" data-index="${idx}">
            <div class="palette-header">
              <button class="collapse-btn" title="Collapse/Expand">‚ñº</button>
              <div class="palette-title">${palette.title}</div>
              <div class="palette-actions">
                <button class="action-btn copy-all-btn" data-index="${idx}">Copy All</button>
                <button class="action-btn secondary save-btn" data-index="${idx}">Save JSON</button>
              </div>
            </div>
            <div class="palette-colors">
              ${colorRows}
            </div>
          </div>
        `;
      }).join('');
      
      // Restore scroll position
      list.scrollTop = scrollPos;
      
      // Add collapse functionality
      list.querySelectorAll('.collapse-btn').forEach(btn => {
        btn.addEventListener('click', (e) => {
          const group = e.target.closest('.palette-group');
          const colors = group.querySelector('.palette-colors');
          const isCollapsed = colors.classList.toggle('hidden');
          e.target.classList.toggle('collapsed', isCollapsed);
        });
      });
      
      // Add individual color copy
      list.querySelectorAll('.copy-btn').forEach(btn => {
        btn.addEventListener('click', async (e) => {
          const className = e.target.dataset.copy;
          await navigator.clipboard.writeText(className);
          
          e.target.textContent = '‚úì Copied';
          e.target.classList.add('copied');
          
          setTimeout(() => {
            e.target.textContent = 'Copy';
            e.target.classList.remove('copied');
          }, 1500);
        });
      });
      
      // Add copy all functionality
      list.querySelectorAll('.copy-all-btn').forEach(btn => {
        btn.addEventListener('click', async (e) => {
          const idx = e.target.dataset.index;
          const palette = palettes[idx];
          const classes = palette.colors.map(hex => {
            const match = findClosest(hex);
            return `${match.family}-${match.shade}`;
          });
          
          await navigator.clipboard.writeText(classes.join(' '));
          
          e.target.textContent = '‚úì Copied!';
          e.target.classList.add('copied');
          
          setTimeout(() => {
            e.target.textContent = 'Copy All';
            e.target.classList.remove('copied');
          }, 1500);
        });
      });
      
      // Add save JSON functionality
      list.querySelectorAll('.save-btn').forEach(btn => {
        btn.addEventListener('click', (e) => {
          const idx = e.target.dataset.index;
          const palette = palettes[idx];
          
          const data = {
            name: palette.title,
            colors: palette.colors.map(hex => {
              const match = findClosest(hex);
              return {
                original: hex,
                tailwind: `${match.family}-${match.shade}`,
                tailwindHex: match.hex
              };
            })
          };
          
          const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
          const url = URL.createObjectURL(blob);
          const a = document.createElement('a');
          a.href = url;
          a.download = `${palette.title.replace(/[^a-z0-9]/gi, '-').toLowerCase()}.json`;
          a.click();
          URL.revokeObjectURL(url);
          
          e.target.textContent = '‚úì Saved';
          setTimeout(() => {
            e.target.textContent = 'Save JSON';
          }, 1500);
        });
      });
    }
    
    // Initial load
    const initialPalettes = await extractPalettes();
    currentPalettes = initialPalettes;
    renderPalettes(currentPalettes);
    
    // Start live updating every 2 seconds
    updateInterval = setInterval(async () => {
      if (isUpdating) return;
      
      isUpdating = true;
      try {
        const newPalettes = await extractPalettes();
        
        // Only update if palettes changed
        if (JSON.stringify(newPalettes) !== JSON.stringify(currentPalettes)) {
          currentPalettes = newPalettes;
          renderPalettes(currentPalettes);
        }
      } catch (err) {
        console.error('Update error:', err);
      }
      isUpdating = false;
    }, 2000);
    
  } catch (err) {
    console.error(err);
    status.textContent = 'Error loading colors';
    list.innerHTML = '<div class="empty"><div class="empty-icon">‚ùå</div><div class="empty-text">Something went wrong. Try refreshing the page.</div></div>';
  }
});

// Cleanup on popup close
window.addEventListener('unload', () => {
  if (updateInterval) {
    clearInterval(updateInterval);
  }
});
